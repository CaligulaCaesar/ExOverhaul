namespace = ex_sr_deposit_events

event = {
	id = ex_sr_deposit_events.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		every_planet = {
			if = {
				limit = { 
					NOT = { has_global_flag = special_resources_rolled }

					has_owner = no
					has_any_strategic_resource = no
				}
				roll_special_resource_deposits_no_sr = yes
				set_global_flag = special_resources_rolled
			} 
			if = {
				limit = { 
					NOT = { has_global_flag = special_resources_rolled }
					is_colonizable = yes
					has_owner = no
					has_any_strategic_resource = yes
				}
				roll_special_resource_deposits_sr_present = yes
				set_global_flag = special_resources_rolled
			}
		}
	}	
}
