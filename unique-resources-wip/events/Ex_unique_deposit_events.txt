namespace = ex_unique_deposit_events

event = {
	id = ex_unique_deposit_events.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		every_planet = {
			if = {
				limit = { 
					AND = {
						is_habitable_planet = yes
						has_unique_deposit = no
						planet_size > 15
						has_owner = no
					}
					
						NOT = { has_planet_flag = ex_tw_modifier_rolled }
					}
				}
				ex_tw_roll_modifier = yes				
				}
			set_planet_flag = ex_tw_modifier_rolled					
			}	
		}
	}
}

event = {
	id = ex_tw_events.2
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		FROMFROM = {
			every_system_planet = {
				if = {
					limit = {
						AND = {
							is_planet_class = pc_nuked
							NOT = { has_planet_flag = ex_tw_modifier_rolled }
						}
					}
					ex_tw_roll_modifier = yes
				}						
				set_planet_flag = ex_tw_modifier_rolled								
			}
		}
	}
}
