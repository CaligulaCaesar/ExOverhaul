namespace = ex_unique_deposit_events

@unique_spawn_chance = 66

event = {
	id = ex_unique_deposit_events.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		random = {
			chance = @unique_spawn_chance
			random_system = {
				limit = { 
					any_planet = { 
						AND = {
							is_colonizable = yes
							planet_size > 15
							has_owner = no	
						}						
					}
				}
				random_system_planet = {
					limit = { 
						AND = {
							is_colonizable = yes
							planet_size > 15
							has_owner = no	
						}	
					}
					random_tile = {
						limit = { 
							has_blocker = no 
							has_deposit = no
							has_building = no		
						}
						add_deposit = "d_sr_ex_unique_1_deposit"
					}
				}
			}
		}
	}
}
